#include <Servo.h>

Servo motorA;
Servo motorAV;
Servo motorB;
Servo motorBV;
Servo motorC;
Servo motorCV;
Servo motorD;
Servo motorDV;

char letra='\0';
bool estado=true;


void setup()
{
    motorA.attach(11);
    motorB.attach(10);
    motorC.attach(9);
    motorD.attach(8);
    motorAV.attach(7);
    motorBV.attach(6);
    motorCV.attach(5);
    motorDV.attach(4);

    pinMode(13, OUTPUT);
    digitalWrite(13, LOW);

    Serial.begin(9600);
    Serial.println("MODULO CONECTADO");
    Serial.print("#");
    motorA.write(0);
    motorB.write(0);
    motorC.write(0);
    motorD.write(0);
    motorAV.write(0);
    motorBV.write(0);
    motorCV.write(0);
    motorDV.write(0);
}

void loop()
{
    if (Serial.available())
    {
        char VarChar;

        VarChar = Serial.read();
///MOTOR A
        if(VarChar == '1')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorA.write(45);
        delay(100);

        }
        if(VarChar == '0')
        {
        digitalWrite(13, LOW);
        delay(100);
        motorA.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }
///MOTOR B
        if(VarChar == '3')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorB.write(45);
        delay(100);

        }

        {
        digitalWrite(13, LOW);
        delay(100);
        motorB.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }
///MOTOR C
        if(VarChar == '5')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorC.write(45);
        delay(100);

        }
        if(VarChar == '4')
        {
        digitalWrite(13, LOW);
        delay(100);
        motorC.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }
///MOTOR D
        if(VarChar == '7')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorD.write(45);
        delay(100);

        }
        if(VarChar == '6')
        {
        digitalWrite(13, LOW);
        delay(100);
        motorD.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }
///MOTOR A-VERTICAL
        if(VarChar == 'a')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorAV.write(45);
        delay(100);

        }
        if(VarChar == 'b')
        {
        digitalWrite(13, LOW);
        delay(100);
        motorAV.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }

///MOTOR B-VERTICAL
        if(VarChar == 'c')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorBV.write(45);
        delay(100);

        }
        if(VarChar == 'd')
        {
        digitalWrite(13, LOW);
        delay(100);
        motorBV.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }
///MOTOR C-VERTICAL
        if(VarChar == 'e')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorCV.write(45);
        delay(100);

        }
        if(VarChar == 'f')
        {
        digitalWrite(13, LOW);
        delay(100);
        motorCV.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }
///MOTOR D-VERTICAL
        if(VarChar == 'g')
        {
        digitalWrite(13, HIGH);
        delay(100);
        motorDV.write(45);
        delay(100);

        }
        if(VarChar == 'h')
        {
        digitalWrite(13, LOW);
        delay(100);
        motorDV.write(-45);
        delay(500);
        Serial.print("LED APAGADO#");
        }



    }
}
